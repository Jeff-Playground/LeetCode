public int gcd(int x, int y){
    if(x<y){
        return gcd(y, x);
    }
    if(y==0){
        return x;
    }
    if(x%2==0 && y%2==0){
        return 2*gcd(x>>1, y>>1);
    } else if(x%2==0){
        return gcd(x>>1, y);
    } else if(y%2==0){
        return gcd(x, y>>1);
    } else{
        return gcd(x-y, y);
    }
}
